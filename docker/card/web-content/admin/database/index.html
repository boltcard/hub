Admin Password

Run these commands to reset the admin password in a database export.

read -s -p "Admin password: " PW && echo
SALT=$(openssl rand -hex 16)
HASH=$(printf '%s' "${SALT}${PW}" | sha256sum | cut -d' ' -f1)
unset PW
sqlite3 cards.db "UPDATE settings SET value='${SALT}' WHERE name='admin_password_salt';"
sqlite3 cards.db "UPDATE settings SET value='${HASH}' WHERE name='admin_password_hash';"
